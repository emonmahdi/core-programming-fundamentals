## **1ï¸âƒ£ Sort an Array using Bubble Sort**  
ğŸ”¹ **Problem:** Implement **Bubble Sort** to sort an array in ascending order.  
ğŸ”¹ **Example:**
```javascript
Input: [64, 34, 25, 12, 22, 11, 90]
Output: [11, 12, 22, 25, 34, 64, 90]
```
ğŸ”¹ **Solution:**
```javascript
function bubbleSort(arr) {
  let n = arr.length;
  for (let i = 0; i < n - 1; i++) {
    for (let j = 0; j < n - i - 1; j++) {
      if (arr[j] > arr[j + 1]) {
        [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]]; // Swap
      }
    }
  }
  return arr;
}

console.log(bubbleSort([64, 34, 25, 12, 22, 11, 90]));
```

## **2ï¸âƒ£ Sort an Array using Quick Sort**  
ğŸ”¹ **Problem:** Implement **Quick Sort** to sort an array in ascending order.  
ğŸ”¹ **Example:**
```javascript
Input: [10, 7, 8, 9, 1, 5]
Output: [1, 5, 7, 8, 9, 10]
```
ğŸ”¹ **Solution:**
```javascript
function quickSort(arr) {
  if (arr.length <= 1) return arr;

  let pivot = arr[arr.length - 1];
  let left = arr.filter(num => num < pivot);
  let right = arr.filter(num => num > pivot);

  return [...quickSort(left), pivot, ...quickSort(right)];
}

console.log(quickSort([10, 7, 8, 9, 1, 5]));
```

## **4ï¸âƒ£ Binary Search in a Sorted Array**  
ğŸ”¹ **Problem:** Implement **Binary Search** to find an element in a sorted array.  
ğŸ”¹ **Example:**
```javascript
Input: arr = [1, 2, 3, 4, 5, 6], target = 4
Output: Index = 3
```
ğŸ”¹ **Solution:**
```javascript
function binarySearch(arr, target) {
  let left = 0, right = arr.length - 1;
  
  while (left <= right) {
    let mid = Math.floor((left + right) / 2);
    
    if (arr[mid] === target) return mid;
    else if (arr[mid] < target) left = mid + 1;
    else right = mid - 1;
  }

  return -1; // Not found
}

console.log(binarySearch([1, 2, 3, 4, 5, 6], 4));
```

## **5ï¸âƒ£ Find the First and Last Occurrence of an Element in a Sorted Array**  
ğŸ”¹ **Problem:** Given a sorted array, find the **first and last occurrence** of a target element.  
ğŸ”¹ **Example:**
```javascript
Input: arr = [5, 7, 7, 8, 8, 10], target = 8
Output: [3, 4]
```
ğŸ”¹ **Solution:**
```javascript
function firstAndLastOccurrence(arr, target) {
  let first = arr.indexOf(target);
  let last = arr.lastIndexOf(target);
  return first === -1 ? [-1, -1] : [first, last];
}

console.log(firstAndLastOccurrence([5, 7, 7, 8, 8, 10], 8));
```

## **6ï¸âƒ£ Find a Peak Element in an Array**  
ğŸ”¹ **Problem:** Find an element that is greater than its neighbors in an array.  
ğŸ”¹ **Example:**
```javascript
Input: [1, 3, 20, 4, 1, 0]
Output: 20
```
ğŸ”¹ **Solution:**
```javascript
function findPeakElement(arr) {
  for (let i = 1; i < arr.length - 1; i++) {
    if (arr[i] > arr[i - 1] && arr[i] > arr[i + 1]) return arr[i];
  }
  return -1;
}

console.log(findPeakElement([1, 3, 20, 4, 1, 0]));
```

## **9ï¸âƒ£ Tower of Hanoi Problem**  
ğŸ”¹ **Problem:** Solve the Tower of Hanoi for `n` disks.  
ğŸ”¹ **Example:**
```javascript

Output:

Move disk 2 from A to B

```
ğŸ”¹ **Solution:**
```javascript

  if (n === 1) {
    console.log(`Move disk 1 from ${fromRod} to ${toRod}`);
    return;
  }
  towerOfHanoi(n - 1, fromRod, auxRod, toRod);
  console.log(`Move disk ${n} from ${fromRod} to ${toRod}`);
  towerOfHanoi(n - 1, auxRod, toRod, fromRod);
}

towerOfHanoi(3, 'A', 'C', 'B');
```
